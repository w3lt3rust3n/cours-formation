Concaténation avec + et \n pour aller à la ligne

let x = 2;
let y = 4;
let z = 5
let a = x + 1; //a stocke 2 + 1 = 3
let b = x + y; //b stocke 2 + 4 = 6
let c = x - y; //c stocke 2 - 4 = -2
let d = x * y; //d stocke 2 * 4 = 8
let e = x / y; //e stocke 2 / 4
let f = z % 3; //f stocke le reste de la division euclidienne de 5 par 3 avec le modulo %
let g = x ** 3; //g stocke 2^3 = 2 * 2 * 2 = 8
/*On affiche les résultats dans une boite d'alerte en utilisant l'opérateur
*de concaténation "+". On retourne à la ligne dans l'affichage avec "\n"*/
alert('a contient : ' + a +
'\nb contient : ' + b +
'\nc contient : ' + c +
'\nd contient : ' + d +
'\ne contient : ' + e +
'\nf contient : ' + f +
'\ng contient : ' + g);
